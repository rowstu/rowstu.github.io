zimport datetime
from datetime import date,timedelta
import sys

def main():
    
    epoch = date(1970,1,1)
    input = (sys.stdin.read().split())

    for t in input:
        d = epoch + timedelta(days=int(t))
        # print(d)
        print(d.strftime('%A'))


if __name__ == "__main__":
    main()
