Game 1: Tower of Terror

Climb the haunted tower before midnight! But watch your pulse rate...

How It Works

Navigate through a haunted tower, encountering increasingly scary creatures. Each scare raises your pulse. If it goes over 150, you die of fright! Reach the top before midnight.

BASIC

10 REM TOWER OF TERROR
20 P = 70: F = 1: T = 9
30 DIM M$(3)
40 M$(1) = "SKELETON": M$(2) = "GHOST"
50 M$(3) = "HEADLESS AXEMAN"
60 PRINT "FLOOR "; F; " - PULSE: "; P
70 PRINT "TIME: "; T; ":00"
80 IF P > 150 THEN PRINT "HEART ATTACK!": END
90 IF T >= 12 THEN PRINT "MIDNIGHT! TRAPPED!": END
100 PRINT "YOU SEE A "; M$(F); "!"
110 P = P + F * 10 + INT(RND(1)*20)
120 INPUT "(G)O UP OR (R)ETREAT"; A$
130 IF A$ = "G" THEN F = F + 1: T = T + 1
140 IF A$ = "R" THEN P = P - 10: T = T + 1
150 IF F > 3 THEN PRINT "TREASURE FOUND!": END
160 GOTO 60

JavaScript

class TowerOfTerror {
    constructor() {
        this.pulse = 70;
        this.floor = 1;
        this.time = 9; // 9:00 PM
        this.monsters = ["SKELETON", "GHOST", "HEADLESS AXEMAN"];
    }

    encounter() {
        const monster = this.monsters[this.floor - 1];
        const scare = this.floor * 10 + Math.floor(Math.random() * 20);
        this.pulse += scare;
        return { monster, scare };
    }

    goUp() {
        this.floor++;
        this.time++;
        return this.checkState();
    }

    retreat() {
        this.pulse -= 10;
        this.time++;
        return this.checkState();
    }

    checkState() {
        if (this.pulse > 150) return { status: 'dead', message: 'Heart attack!' };
        if (this.time >= 12) return { status: 'trapped', message: 'Midnight! Trapped forever!' };
        if (this.floor > 3) return { status: 'won', message: 'Treasure found!' };
        return { status: 'playing' };
    }
}

Game 2: Skulls of the Pyramid

Drop skulls onto numbered stones to match the Spirit Master's target!

How It Works

The Spirit Master gives you a target number. Drop exactly 5 skulls onto numbered columns (1-9). The sum must equal the target to break the curse!

BASIC

10 REM SKULLS OF THE PYRAMID
20 T = INT(RND(1)*30) + 10
30 PRINT "TARGET: "; T
40 S = 0
50 FOR K = 1 TO 5
60 PRINT "SKULL "; K; " OF 5"
70 INPUT "DROP ON COLUMN (1-9)"; C
80 IF C < 1 OR C > 9 THEN GOTO 70
90 S = S + C
100 PRINT "TOTAL SO FAR: "; S
110 NEXT K
120 IF S = T THEN PRINT "CURSE BROKEN!"
130 IF S <> T THEN PRINT "FAILED! TARGET WAS "; T

JavaScript

class SkullsGame {
    constructor() {
        this.target = Math.floor(Math.random() * 30) + 10;
        this.skulls = 5;
        this.total = 0;
        this.drops = [];
    }

    dropSkull(column) {
        if (column < 1 || column > 9) return { error: 'Invalid column!' };
        if (this.skulls <= 0) return { error: 'No skulls left!' };

        this.skulls--;
        this.total += column;
        this.drops.push(column);

        if (this.skulls === 0) {
            return this.checkResult();
        }

        return {
            skulls: this.skulls,
            total: this.total,
            target: this.target
        };
    }

    checkResult() {
        if (this.total === this.target) {
            return { won: true, message: 'CURSE BROKEN!' };
        }
        return { won: false, message: `Failed! Got ${this.total}, needed ${this.target}` };
    }
}

Game 3: Monster Wrestling

Fight monsters using mental math! Calculate if you can overpower them!

How It Works

Each round, a monster appears with a SIZE and DISTANCE. Calculate SIZE √ó DISTANCE mentally. If your EFFORT (guess) exceeds this number, you win the round. Survive 12 rounds!

BASIC

10 REM MONSTER WRESTLING
20 O = 100: A = 50
30 FOR R = 1 TO 12
40 S = INT(RND(1)*10) + R
50 D = INT(RND(1)*5) + 1
60 P = S * D
70 PRINT "ROUND "; R
80 PRINT "MONSTER SIZE: "; S; " DISTANCE: "; D
90 INPUT "YOUR EFFORT"; E
100 IF E >= P THEN PRINT "YOU PUSH BACK!": A = A + 5
110 IF E < P THEN PRINT "OVERWHELMED!": O = O - 10
120 IF O <= 0 THEN PRINT "YOU COLLAPSED!": END
130 NEXT R
140 PRINT "YOU SURVIVED THE ARENA!"

JavaScript

class MonsterWrestling {
    constructor() {
        this.oxygen = 100;
        this.adrenalin = 50;
        this.round = 0;
        this.maxRounds = 12;
    }

    nextRound() {
        this.round++;
        const size = Math.floor(Math.random() * 10) + this.round;
        const distance = Math.floor(Math.random() * 5) + 1;
        const power = size * distance;
        return { size, distance, power };
    }

    fight(effort, actualPower) {
        if (effort >= actualPower) {
            this.adrenalin += 5;
            return { success: true, message: 'You push the monster back!' };
        }
        this.oxygen -= 10;
        return { success: false, message: 'Overwhelmed!' };
    }

    checkState() {
        if (this.oxygen <= 0) return { status: 'lost', message: 'You collapsed!' };
        if (this.round >= this.maxRounds) return { status: 'won', message: 'Survived!' };
        return { status: 'playing' };
    }
}

Game 4: Jaws

You ARE the predator! Hunt people while avoiding the hunter!

How It Works

You play as Jaws (J) on a grid. Hunt people (P) by moving onto them. But a Hunter (H) is after you! Move with A/Z/N/M keys. Eat as many as you can before being caught!

BASIC

10 REM JAWS
20 DIM G(10,10)
30 JX = 5: JY = 5: HX = 1: HY = 1
40 S = 0: M = 0
50 REM Place people randomly
60 FOR I = 1 TO 5
70 G(INT(RND(1)*10)+1, INT(RND(1)*10)+1) = 1
80 NEXT I
90 REM Display and move
100 K$ = INKEY$
110 IF K$ = "A" THEN JX = JX - 1
120 IF K$ = "Z" THEN JX = JX + 1
130 IF K$ = "N" THEN JY = JY - 1
140 IF K$ = "M" THEN JY = JY + 1
150 REM Check for eating
160 IF G(JX,JY) = 1 THEN S = S + 1: G(JX,JY) = 0
170 REM Move hunter toward jaws
180 IF HX < JX THEN HX = HX + 1
190 IF HX > JX THEN HX = HX - 1
200 IF HY < JY THEN HY = HY + 1
210 IF HY > JY THEN HY = HY - 1
220 IF HX = JX AND HY = JY THEN PRINT "CAUGHT!": END
230 GOTO 100

JavaScript

class JawsGame {
    constructor(size = 10) {
        this.size = size;
        this.jaws = { x: 5, y: 5 };
        this.hunter = { x: 0, y: 0 };
        this.people = [];
        this.score = 0;
        this.moves = 0;

        // Spawn people
        for (let i = 0; i < 5; i++) {
            this.people.push({
                x: Math.floor(Math.random() * size),
                y: Math.floor(Math.random() * size)
            });
        }
    }

    moveJaws(dx, dy) {
        const newX = Math.max(0, Math.min(this.size-1, this.jaws.x + dx));
        const newY = Math.max(0, Math.min(this.size-1, this.jaws.y + dy));
        this.jaws.x = newX;
        this.jaws.y = newY;
        this.moves++;

        this.checkEat();
        this.moveHunter();
        return this.checkCaught();
    }

    checkEat() {
        this.people = this.people.filter(p => {
            if (p.x === this.jaws.x && p.y === this.jaws.y) {
                this.score++;
                return false;
            }
            return true;
        });
    }

    moveHunter() {
        if (this.hunter.x < this.jaws.x) this.hunter.x++;
        else if (this.hunter.x > this.jaws.x) this.hunter.x--;
        if (this.hunter.y < this.jaws.y) this.hunter.y++;
        else if (this.hunter.y > this.jaws.y) this.hunter.y--;
    }

    checkCaught() {
        return this.hunter.x === this.jaws.x && this.hunter.y === this.jaws.y;
    }
}

Game 6: Micropuzzle

A text adventure! Wake up in a strange place and figure out what's happening...

How It Works

Type two-word commands like "GO NORTH", "EXAMINE DOOR", "TAKE KEY". Explore rooms, collect items, and solve the mystery. This is the most complex game in the book - a full text parser!

BASIC

10 REM MICROPUZZLE - TEXT ADVENTURE
20 DIM R$(10): DIM O$(5)
30 R = 1: REM Current room
40 REM Room descriptions
50 R$(1) = "A DARK CELL"
60 R$(2) = "A CORRIDOR"
70 R$(3) = "THE EXIT"
80 PRINT "YOU ARE IN "; R$(R)
90 INPUT "WHAT DO YOU DO"; A$
100 REM Parse command
110 V$ = LEFT$(A$, 2)
120 IF V$ = "GO" AND R = 1 THEN R = 2
130 IF V$ = "GO" AND R = 2 THEN R = 3
140 IF R = 3 THEN PRINT "YOU ESCAPED!": END
150 GOTO 80

JavaScript

class Micropuzzle {
    constructor() {
        this.currentRoom = 0;
        this.inventory = [];
        this.rooms = [
            { name: "Dark Cell", description: "You wake up in a dark cell. There's a door to the NORTH.", exits: { north: 1 } },
            { name: "Corridor", description: "A long corridor. Doors SOUTH and EAST.", exits: { south: 0, east: 2 } },
            { name: "Exit", description: "Freedom! Bright light floods in.", exits: {} }
        ];
    }

    parseCommand(input) {
        const parts = input.toLowerCase().split(' ');
        const verb = parts[0];
        const noun = parts.slice(1).join(' ');

        switch(verb) {
            case 'go': return this.go(noun);
            case 'look': return this.look();
            case 'take': return this.take(noun);
            case 'examine': return this.examine(noun);
            default: return "I don't understand.";
        }
    }

    go(direction) {
        const room = this.rooms[this.currentRoom];
        if (room.exits[direction] !== undefined) {
            this.currentRoom = room.exits[direction];
            return this.look();
        }
        return "You can't go that way.";
    }

    look() {
        return this.rooms[this.currentRoom].description;
    }
}

Python

class Micropuzzle:
    def __init__(self):
        self.current_room = 0
        self.inventory = []
        self.rooms = [
            {"name": "Dark Cell", "desc": "A dark cell. Door NORTH.", "exits": {"north": 1}},
            {"name": "Corridor", "desc": "A corridor. SOUTH or EAST.", "exits": {"south": 0, "east": 2}},
            {"name": "Exit", "desc": "Freedom!", "exits": {}}
        ]

    def parse_command(self, input_str):
        parts = input_str.lower().split()
        verb = parts[0] if parts else ""
        noun = " ".join(parts[1:])

        if verb == "go":
            return self.go(noun)
        elif verb == "look":
            return self.look()
        return "I don't understand."

    def go(self, direction):
        room = self.rooms[self.current_room]
        if direction in room["exits"]:
            self.current_room = room["exits"][direction]
            return self.look()
        return "You can't go that way."

    def look(self):
        return self.rooms[self.current_room]["desc"]
