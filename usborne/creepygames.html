// Build suspense with delays
async function spookySequence() {
    console.log("THE GHOST APPROACHES...");
    await delay(2000);
    console.log("BOO!");

    // Random delay for unpredictability
    const d = Math.random() * 1000 + 500;
    await delay(d);
}

function delay(ms) {
    return new Promise(r => setTimeout(r, ms));
}

Game 1: Computer Nightmare

Numbers attack from everywhere! Hit them before they overwhelm you!

How It Works

Random numbers flash on screen. Press the matching number key as fast as you can. Miss too many and the nightmare consumes you!

BASIC

10 REM COMPUTER NIGHTMARE
20 S = 0: M = 0
30 FOR R = 1 TO 20
40 N = INT(RND(1)*9) + 1
50 PRINT "NUMBER: "; N
60 FOR T = 1 TO 500
70 K$ = INKEY$
80 IF K$ = "" THEN NEXT T: M = M + 1: GOTO 100
90 IF VAL(K$) = N THEN S = S + 1
100 NEXT R
110 PRINT "SCORE: "; S; " MISSED: "; M

JavaScript

class ComputerNightmare {
    constructor() {
        this.score = 0;
        this.missed = 0;
        this.round = 0;
        this.currentNumber = 0;
        this.waiting = false;
    }

    nextNumber() {
        this.round++;
        this.currentNumber = Math.floor(Math.random() * 9) + 1;
        this.waiting = true;
        return this.currentNumber;
    }

    pressKey(key) {
        if (!this.waiting) return;
        this.waiting = false;
        if (parseInt(key) === this.currentNumber) {
            this.score++;
            return true;
        }
        return false;
    }

    timeout() {
        if (this.waiting) {
            this.missed++;
            this.waiting = false;
        }
    }
}

        return { die1: this.die1, die2: this.die2, target: this.target };

        if (num1 < 1 || num1 > 9 || num2 < 1 || num2 > 9) {
            return { valid: false, message: 'Numbers must be 1-9!' };

        if (num1 + num2 === this.target) {
            return { correct: true, message: 'The wizard is impressed!' };
        }
        return { correct: false, message: 'The wizard casts a spell on you!' };
    }
}

class GhostGuzzler {

Game 4: Spiderwoman

Guess the Spiderwoman's secret letter before she turns you into a fly!

How It Works

The Spiderwoman thinks of a letter A-Z. Give her words - she'll tell you if they contain her secret letter. Guess the letter to escape!

30 PRINT "SPIDERWOMAN THINKS OF A LETTER..."
40 FOR G = 1 TO 10

60 F = 0
70 FOR I = 1 TO LEN(W$)
80 IF MID$(W$, I, 1) = S$ THEN F = 1
90 NEXT I
100 IF F = 1 THEN PRINT "YES, IT CONTAINS MY LETTER"
110 IF F = 0 THEN PRINT "NO, IT DOESN'T"

JavaScript

class Spiderwoman {
    constructor() {
        this.secretLetter = String.fromCharCode(
            Math.floor(Math.random() * 26) + 65
        );
        this.guesses = 10;
    }

    checkWord(word) {
        word = word.toUpperCase();
        return word.includes(this.secretLetter);
    }

    guessLetter(letter) {
        this.guesses--;
        letter = letter.toUpperCase();

        if (letter === this.secretLetter) {
            return { correct: true, message: 'You escape!' };
        }

        if (this.guesses <= 0) {
            return {
                correct: false,
                gameOver: true,
                message: `You're a fly! Letter was ${this.secretLetter}`
            };
        }

        return { correct: false, guessesLeft: this.guesses };
    }
}

JavaScript

class GraveDigger {
    constructor(size = 10) {
        this.size = size;
        this.player = { x: 0, y: 0 };
        this.skeleton = { x: size-1, y: size-1 };
        this.exit = { x: size-1, y: 0 };
        this.holes = new Set();
        this.gameOver = false;
    }

    move(direction) {
        const moves = {
            'w': { x: 0, y: -1 }, 'a': { x: -1, y: 0 },
            's': { x: 0, y: 1 },  'd': { x: 1, y: 0 }
        };
        const m = moves[direction.toLowerCase()];
        if (!m) return;

        const newX = this.player.x + m.x;
        const newY = this.player.y + m.y;

        if (newX >= 0 && newX < this.size && newY >= 0 && newY < this.size) {
            this.player.x = newX;
            this.player.y = newY;
        }

        this.moveSkeleton();
        this.checkState();
    }

    dig() {
        this.holes.add(`${this.player.x},${this.player.y}`);
        this.moveSkeleton();
        this.checkState();
    }

    moveSkeleton() {
        // Move toward player
        if (this.skeleton.x < this.player.x) this.skeleton.x++;
        else if (this.skeleton.x > this.player.x) this.skeleton.x--;
        if (this.skeleton.y < this.player.y) this.skeleton.y++;
        else if (this.skeleton.y > this.player.y) this.skeleton.y--;
    }

    checkState() {
        // Check if skeleton in hole
        if (this.holes.has(`${this.skeleton.x},${this.skeleton.y}`)) {
            this.skeleton = { x: -10, y: -10 }; // Trapped!
        }
        // Check win/lose
        if (this.player.x === this.exit.x && this.player.y === this.exit.y) {
            this.gameOver = true;
            return 'escaped';
        }
        if (this.player.x === this.skeleton.x && this.player.y === this.skeleton.y) {
            this.gameOver = true;
            return 'caught';
        }
        return 'playing';
    }
}

class GraveDigger:
    def __init__(self, size=10):
        self.size = size
        self.player = [0, 0]
        self.skeleton = [size-1, size-1]
        self.exit = [size-1, 0]
        self.holes = set()

    def move(self, direction):
        moves = {'w': (0,-1), 'a': (-1,0), 's': (0,1), 'd': (1,0)}
        if direction.lower() not in moves:
            return
        dx, dy = moves[direction.lower()]
        new_x = self.player[0] + dx
        new_y = self.player[1] + dy
        if 0 <= new_x < self.size and 0 <= new_y < self.size:
            self.player = [new_x, new_y]
        self.move_skeleton()

    def dig(self):
        self.holes.add(tuple(self.player))
        self.move_skeleton()

JavaScript

class Seance {
    constructor() {
        this.sequence = [];
        this.level = 4;
        this.currentIndex = 0;
    }

    generateSequence() {
        this.sequence = [];
        for (let i = 0; i < this.level; i++) {
            this.sequence.push(
                String.fromCharCode(Math.floor(Math.random() * 26) + 65)
            );
        }
        this.currentIndex = 0;
        return this.sequence;
    }

    checkLetter(letter) {
        if (letter.toUpperCase() !== this.sequence[this.currentIndex]) {
            return { correct: false, message: 'Spirit angered!' };
        }

        this.currentIndex++;
        if (this.currentIndex >= this.sequence.length) {
            this.level++;
            return { correct: true, complete: true };
        }
        return { correct: true, complete: false };
    }
}

import random

class Seance:
    def __init__(self):
        self.sequence = []
        self.level = 4
        self.current_index = 0

    def generate_sequence(self):
        self.sequence = [
            chr(random.randint(65, 90))
            for _ in range(self.level)
        ]
        self.current_index = 0
        return self.sequence

    def check_letter(self, letter):
        if letter.upper() != self.sequence[self.current_index]:
            return {"correct": False}

        self.current_index += 1
        if self.current_index >= len(self.sequence):
            self.level += 1
            return {"correct": True, "complete": True}
        return {"correct": True, "complete": False}
