Physics Simulation

Moonlander and other games use real physics formulas for gravity and velocity.

BASIC

10 REM PHYSICS SIMULATION
20 H = 500: V = 0: G = 1.6
30 REM H=height, V=velocity, G=gravity
40 INPUT "FUEL BURN (0-30)"; F
50 REM Update velocity and height
60 V = V + G - F/5
70 H = H - V
80 IF H <= 0 THEN GOSUB 200
90 GOTO 40
200 REM LANDING CHECK
210 IF V > 5 THEN PRINT "CRASH!"
220 IF V <= 5 THEN PRINT "LANDED!"
230 RETURN

class PhysicsSimulation {
    constructor() {
        this.height = 500;
        this.velocity = 0;
        this.gravity = 1.6;
    }

    update(fuelBurn) {
        // Thrust reduces velocity
        this.velocity += this.gravity - fuelBurn / 5;
        this.height -= this.velocity;

        if (this.height <= 0) {
            return this.checkLanding();
        }
        return { landed: false };
    }

    checkLanding() {
        if (this.velocity > 5) {
            return { landed: true, crashed: true };
        }
        return { landed: true, crashed: false };
    }
}

Game 1: Starship Takeoff

You've crash-landed on an alien planet. Guess the correct force to escape!

How It Works

The planet has random gravity (1-20). You need to guess the exact takeoff force (1-40). Too low won't lift off, too high triggers the fail-safe. You have 10 attempts!

BASIC

10 REM STARSHIP TAKEOFF
20 G = INT(RND(1)*20) + 1
30 F = INT(RND(1)*40) + 1
40 PRINT "CRASHED ON ALIEN WORLD!"
50 PRINT "GUESS THE TAKEOFF FORCE"
60 FOR A = 1 TO 10
70 INPUT "FORCE (1-40)"; X
80 IF X = F THEN PRINT "TAKEOFF!": END
90 IF X < F THEN PRINT "NOT ENOUGH THRUST"
100 IF X > F THEN PRINT "FAIL-SAFE TRIGGERED"
110 NEXT A
120 PRINT "STRANDED FOREVER!"

JavaScript

class StarshipTakeoff {
    constructor() {
        this.gravity = Math.floor(Math.random() * 20) + 1;
        this.requiredForce = Math.floor(Math.random() * 40) + 1;
        this.attempts = 10;
    }

    tryTakeoff(force) {
        this.attempts--;

        if (force === this.requiredForce) {
            return { success: true, message: 'TAKEOFF! You escaped!' };
        }

        if (this.attempts <= 0) {
            return { success: false, message: 'Stranded forever!', gameOver: true };
        }

        const hint = force < this.requiredForce
            ? 'Not enough thrust!'
            : 'Fail-safe triggered! Too much force!';

        return { success: false, hint, attemptsLeft: this.attempts };
    }
}

import random

class StarshipTakeoff:
    def __init__(self):
        self.gravity = random.randint(1, 20)
        self.required_force = random.randint(1, 40)
        self.attempts = 10

    def try_takeoff(self, force):
        self.attempts -= 1

        if force == self.required_force:
            return {"success": True, "message": "TAKEOFF!"}

        if self.attempts <= 0:
            return {"game_over": True}

        hint = "Not enough thrust!" if force < self.required_force \
               else "Fail-safe triggered!"
        return {"success": False, "hint": hint}

Game 2: Moonlander

Control your lunar module descent. Land safely before you run out of fuel!

How It Works

You start at height 500 with 150 fuel units. Each turn, choose fuel burn (0-30). Gravity pulls you down, fuel thrust slows you. Land with velocity under 5 for a safe landing!

BASIC

10 REM MOONLANDER
20 H = 500: V = 0: F = 150
30 PRINT "HEIGHT:";H;" VELOCITY:";V;" FUEL:";F
40 IF H <= 0 THEN GOTO 100
50 INPUT "FUEL BURN (0-30)"; B
60 IF B > F THEN B = F
70 F = F - B
80 V = V + 1.6 - B/5
90 H = H - V: GOTO 30
100 REM CHECK LANDING
110 IF V > 5 THEN PRINT "CRASHED!"
120 IF V <= 5 AND V > 2 THEN PRINT "OK LANDING"
130 IF V <= 2 THEN PRINT "PERFECT LANDING!"

JavaScript

class Moonlander {
    constructor() {
        this.height = 500;
        this.velocity = 0;
        this.fuel = 150;
        this.gravity = 1.6;
        this.landed = false;
    }

    burn(amount) {
        if (this.landed) return null;

        // Cap at available fuel
        const burn = Math.min(amount, this.fuel);
        this.fuel -= burn;

        // Update physics
        this.velocity += this.gravity - burn / 5;
        this.height -= this.velocity;

        if (this.height <= 0) {
            this.landed = true;
            return this.checkLanding();
        }

        return {
            height: Math.round(this.height),
            velocity: this.velocity.toFixed(1),
            fuel: this.fuel
        };
    }

    checkLanding() {
        if (this.velocity > 5) return { crashed: true };
        if (this.velocity > 2) return { rating: 'OK LANDING' };
        return { rating: 'PERFECT LANDING!' };
    }
}

Elron the Evil Alien hides in 3D space. Find him with your space bombs!

How It Works

The alien hides at random X, Y, Z coordinates (0-10 each). Drop bombs at specific coordinates and get directional hints: NORTH/SOUTH, EAST/WEST, TOO FAR/NOT FAR ENOUGH!

30 AZ = INT(RND(1)*10)
40 FOR B = 1 TO 4
50 INPUT "X COORDINATE"; X
60 INPUT "Y COORDINATE"; Y
70 INPUT "DISTANCE"; Z
80 IF X=AX AND Y=AY AND Z=AZ THEN 150
90 IF X < AX THEN PRINT "EAST"
100 IF X > AX THEN PRINT "WEST"
110 IF Y < AY THEN PRINT "NORTH"
120 IF Y > AY THEN PRINT "SOUTH"
130 IF Z < AZ THEN PRINT "NOT FAR ENOUGH"
140 IF Z > AZ THEN PRINT "TOO FAR": NEXT B

150 PRINT "ALIEN DESTROYED!"

class EvilAlien {
    constructor() {
        this.alienX = Math.floor(Math.random() * 10);
        this.alienY = Math.floor(Math.random() * 10);
        this.alienZ = Math.floor(Math.random() * 10);
        this.bombs = 4;
    }

    dropBomb(x, y, z) {
        this.bombs--;

        if (x === this.alienX && y === this.alienY && z === this.alienZ) {
            return { hit: true, message: 'ALIEN DESTROYED!' };
        }

        const hints = [];
        if (x < this.alienX) hints.push('EAST');
        if (x > this.alienX) hints.push('WEST');
        if (y < this.alienY) hints.push('NORTH');
        if (y > this.alienY) hints.push('SOUTH');
        if (z < this.alienZ) hints.push('NOT FAR ENOUGH');
        if (z > this.alienZ) hints.push('TOO FAR');

        return { hit: false, hints, bombsLeft: this.bombs };
    }
}

Game 4: Asteroid Belt

Navigate through asteroids! Press the matching number to destroy them!

How It Works

Asteroids appear as groups of 1-9 stars. Quickly press the number key matching how many stars you see to destroy the asteroid. 10 asteroids per game!

BASIC

10 REM ASTEROID BELT
20 S = 0
30 FOR A = 1 TO 10
40 CLS
50 N = INT(RND(1)*9) + 1
60 REM Display N stars
70 FOR I = 1 TO N: PRINT "*"; : NEXT I
80 PRINT
90 FOR T = 1 TO 500
100 K$ = INKEY$
110 IF K$ = "" THEN NEXT T: GOTO 140
120 IF VAL(K$) = N THEN S = S + 1: PRINT "HIT!"
130 GOTO 150
140 PRINT "MISSED!"
150 FOR D = 1 TO 200: NEXT D
160 NEXT A
170 PRINT "SCORE: "; S; " / 10"

JavaScript

class AsteroidBelt {
    constructor() {
        this.score = 0;
        this.asteroid = 0;
        this.totalAsteroids = 10;
        this.currentCount = 0;
        this.waiting = false;
    }

    nextAsteroid() {
        this.asteroid++;
        // Random 1-9 stars
        this.currentCount = Math.floor(Math.random() * 9) + 1;
        this.waiting = true;
        return this.currentCount;
    }

    shoot(number) {
        if (!this.waiting) return null;
        this.waiting = false;

        if (number === this.currentCount) {
            this.score++;
            return { hit: true };
        }
        return { hit: false, correct: this.currentCount };
    }

    isGameOver() {
        return this.asteroid >= this.totalAsteroids;
    }
}

Game 5: Space Mines

Lead a mining colony on Planet Astron. Manage resources to survive 10 years!

How It Works

Each year: sell ore at fluctuating prices, buy/sell mines, purchase food. Balance money, food, people, and satisfaction. Survive 10 years to win!

BASIC

10 REM SPACE MINES
20 M = 1000: P = 100: L = 5: S = 50
30 REM M=money, P=people, L=mines, S=satisfaction
40 FOR Y = 1 TO 10
50 PRINT "YEAR "; Y
60 PRINT "MONEY: $"; M; " PEOPLE: "; P
70 PRINT "MINES: "; L; " SATISFACTION: "; S; "%"
80 C = L * 10: REM Ore produced
90 PR = INT(RND(1)*20) + 10: REM Price
100 PRINT "ORE PRODUCED: "; C; " PRICE: $"; PR
110 INPUT "SELL HOW MUCH ORE"; SO
120 M = M + SO * PR
130 INPUT "BUY FOOD (NEED 1/PERSON, COST $5)"; FO
140 M = M - FO * 5
150 IF FO < P THEN S = S - 10: P = P - 5
160 IF S < 20 THEN PRINT "REVOLT!": END
170 NEXT Y
180 PRINT "COLONY SURVIVED!"

JavaScript

class SpaceMines {
    constructor() {
        this.money = 1000;
        this.people = 100;
        this.mines = 5;
        this.satisfaction = 50;
        this.year = 0;
        this.ore = 0;
        this.orePrice = 0;
    }

    startYear() {
        this.year++;
        this.ore = this.mines * 10;
        this.orePrice = Math.floor(Math.random() * 20) + 10;
        return {
            year: this.year,
            ore: this.ore,
            price: this.orePrice
        };
    }

    sellOre(amount) {
        const sold = Math.min(amount, this.ore);
        this.money += sold * this.orePrice;
        this.ore -= sold;
    }

    buyFood(amount) {
        const cost = amount * 5;
        if (cost > this.money) return false;
        this.money -= cost;

        if (amount < this.people) {
            this.satisfaction -= 10;
            this.people -= 5;
        }
        return true;
    }

    endYear() {
        if (this.satisfaction < 20) {
            return { revolt: true };
        }
        if (this.year >= 10) {
            return { won: true };
        }
        return { continue: true };
    }
}

    def __init__(self):
        self.money = 1000
        self.people = 100
        self.mines = 5

        self.ore = self.mines * 10

            "year": self.year,
            "ore": self.ore,

        self.money += sold * self.ore_price

        if cost > self.money:
            return False
        self.money -= cost
        if amount < self.people:
            self.satisfaction -= 10
            self.people -= 5
        return True

Urgent rescue mission! Allocate energy to engines, life support, and shields!

How It Works

You have 100 energy units total. Distribute between engines (speed), life support (survival), and shields (protection). Random events test your choices!

20 PRINT "ALLOCATE 100 ENERGY UNITS"
30 INPUT "ENGINES"; E
40 INPUT "LIFE SUPPORT"; L
50 INPUT "SHIELDS"; S
60 IF E+L+S > 100 THEN PRINT "OVER 100!": GOTO 20
70 REM RANDOM EVENTS
80 R = INT(RND(1)*3) + 1
90 ON R GOSUB 200, 300, 400
100 PRINT "MISSION ";
110 IF OK = 1 THEN PRINT "SUCCESS!"
120 IF OK = 0 THEN PRINT "FAILED!"
130 END
200 REM ASTEROID STORM
210 IF S > 30 THEN OK = 1: RETURN
220 OK = 0: PRINT "SHIELDS FAILED!": RETURN

310 IF E > 40 THEN OK = 1: RETURN

400 REM LIFE SUPPORT FAILURE

JavaScript

class SpaceRescue {
    constructor() {
        this.totalEnergy = 100;
        this.engines = 0;
        this.lifeSupport = 0;
        this.shields = 0;
    }

    allocate(engines, lifeSupport, shields) {
        if (engines + lifeSupport + shields > this.totalEnergy) {
            return { error: 'Over 100 units!' };
        }
        this.engines = engines;
        this.lifeSupport = lifeSupport;
        this.shields = shields;
        return { success: true };
    }

    runMission() {
        const events = [
            { name: 'Asteroid Storm', check: () => this.shields > 30, fail: 'Shields failed!' },
            { name: 'Engine Trouble', check: () => this.engines > 40, fail: 'Too slow!' },
            { name: 'Life Support Failure', check: () => this.lifeSupport > 25, fail: 'Crew perished!' }
        ];

        const event = events[Math.floor(Math.random() * events.length)];

        if (event.check()) {
            return { success: true, event: event.name };
        }
        return { success: false, event: event.name, message: event.fail };
    }
}

import random

class SpaceRescue:
    def __init__(self):
        self.total_energy = 100
        self.engines = 0
        self.life_support = 0
        self.shields = 0

    def allocate(self, engines, life_support, shields):
        total = engines + life_support + shields
        if total > self.total_energy:
            return {"error": "Over 100 units!"}
        self.engines = engines
        self.life_support = life_support
        self.shields = shields
        return {"success": True}

    def run_mission(self):
        events = [
            ("Asteroid Storm", self.shields > 30),
            ("Engine Trouble", self.engines > 40),
            ("Life Support Failure", self.life_support > 25)
        ]
        event = random.choice(events)
        return {"event": event[0], "survived": event[1]}
