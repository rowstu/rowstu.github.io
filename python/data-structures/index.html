Python Data Structures

Lists

Ordered, mutable, allows duplicates.

fruits = ["apple", "banana", "cherry"]

# Access
fruits[0]     # 'apple'
fruits[-1]    # 'cherry'
fruits[1:3]   # ['banana', 'cherry']

# Modify
fruits.append("date")           # add to end
fruits.insert(1, "blueberry")   # insert at index
fruits.extend(["elderberry"])   # add multiple
fruits.remove("banana")         # remove by value
del fruits[0]                   # remove by index
popped = fruits.pop()           # remove & return last
popped = fruits.pop(0)          # remove & return index 0

# Search
"apple" in fruits    # True
fruits.index("cherry") # returns index or raises ValueError
fruits.count("apple")  # count occurrences

# Sort
fruits.sort()                   # in-place, ascending
fruits.sort(reverse=True)       # in-place, descending
sorted_copy = sorted(fruits)    # returns new list
fruits.reverse()                # reverse in-place

# Other
len(fruits)   # length
fruits.copy() # shallow copy
fruits.clear() # empty the list

# Unpack
a, b, c = [1, 2, 3]
first, *rest = [1, 2, 3, 4]  # first=1, rest=[2,3,4]

Tuples

Ordered, immutable, allows duplicates. Faster than lists for iteration.

coords = (10, 20)
single = (42,)       # note trailing comma for single element

x, y = coords        # unpack
x = coords[0]        # index access
len(coords)          # 2

# Tuples as keys in dicts (lists cannot be)
locations = {(51.5, -0.1): "London", (48.8, 2.3): "Paris"}

# Named tuples (more readable)
from collections import namedtuple
Point = namedtuple('Point', ['x', 'y'])
p = Point(3, 4)
p.x    # 3
p.y    # 4

Dictionaries

Key-value pairs, ordered (Python 3.7+), mutable.

person = {"name": "Alice", "age": 30, "city": "London"}

# Access
person["name"]               # 'Alice'
person.get("name")           # 'Alice' (safe, returns None if missing)
person.get("email", "n/a")   # 'n/a' (default value)

# Modify
person["age"] = 31           # update
person["email"] = "a@b.com"  # add new key
del person["city"]           # delete key
popped = person.pop("age")   # remove & return
person.setdefault("role", "user")  # set only if missing

# Iterate
for key in person:
    print(key)

for key, val in person.items():
    print(key, val)

for key in person.keys():   ...
for val in person.values(): ...

# Merge (Python 3.9+)
merged = dict1 | dict2

# Dict comprehension
squares = {n: n**2 for n in range(1, 6)}
# {1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

# Useful
len(person)          # number of keys
"name" in person     # True (checks keys)
person.copy()        # shallow copy

# defaultdict — never raises KeyError
from collections import defaultdict
counts = defaultdict(int)
counts["x"] += 1   # safe, defaults to 0

# Counter — count occurrences
from collections import Counter
c = Counter("abracadabra")
c.most_common(3)    # [('a', 5), ('b', 2), ('r', 2)]

Sets

Unordered, unique values, mutable. No indexing.

s = {1, 2, 3, 2, 1}   # {1, 2, 3} — duplicates removed
s = set([1, 2, 2, 3])  # same

# Add / remove
s.add(4)
s.discard(99)    # safe (no error if missing)
s.remove(2)      # raises KeyError if missing
s.pop()          # remove arbitrary element

# Set operations
a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

a | b      # union:        {1, 2, 3, 4, 5, 6}
a & b      # intersection: {3, 4}
a - b      # difference:   {1, 2}
a ^ b      # symmetric diff: {1, 2, 5, 6}

a.issubset(b)     # False
a.issuperset(b)   # False
a.isdisjoint(b)   # False (they share 3, 4)

# Fast membership test (O(1) vs O(n) for lists)
1000 in some_set   # very fast

List Comprehensions

# Basic
squares = [x**2 for x in range(10)]

# With filter
evens = [x for x in range(20) if x % 2 == 0]

# Nested
pairs = [(x, y) for x in range(3) for y in range(3)]

# Transform strings
words = ["hello", "world"]
upper = [w.upper() for w in words]

# Dict comprehension
word_lengths = {w: len(w) for w in words}

# Set comprehension
unique_lengths = {len(w) for w in words}

# Generator expression (lazy, no list built in memory)
total = sum(x**2 for x in range(1000000))

Sorting

data = [3, 1, 4, 1, 5, 9, 2, 6]

sorted(data)               # ascending
sorted(data, reverse=True) # descending

# Sort by key
words = ["banana", "apple", "fig", "cherry"]
sorted(words, key=len)          # by length
sorted(words, key=str.lower)    # case-insensitive

# Sort list of dicts
people = [{"name": "Bob", "age": 25}, {"name": "Alice", "age": 30}]
sorted(people, key=lambda p: p["age"])
sorted(people, key=lambda p: p["name"])

# Multiple sort keys
sorted(people, key=lambda p: (p["age"], p["name"]))
