Python Functions

Defining Functions

def greet(name):
    return f"Hello, {name}!"

greet("Alice")   # 'Hello, Alice!'

# Default arguments
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

greet("Bob")              # 'Hello, Bob!'
greet("Bob", "Hi")        # 'Hi, Bob!'
greet("Bob", greeting="Hey")  # 'Hey, Bob!'

*args and **kwargs

# *args — variable positional args (tuple)
def total(*nums):
    return sum(nums)

total(1, 2, 3)     # 6
total(1, 2, 3, 4)  # 10

# **kwargs — variable keyword args (dict)
def show(**info):
    for key, val in info.items():
        print(f"{key}: {val}")

show(name="Alice", age=30, city="London")

# Combining
def func(required, *args, **kwargs):
    print(required, args, kwargs)

func("hello", 1, 2, x=3, y=4)
# hello (1, 2) {'x': 3, 'y': 4}

# Unpacking when calling
nums = [1, 2, 3]
print(*nums)          # 1 2 3  (unpack list as positional)

config = {"sep": ", ", "end": "!\n"}
print("a", "b", "c", **config)   # a, b, c!

Lambda Functions

Anonymous one-liner functions.

square = lambda x: x ** 2
square(5)    # 25

# Common with sorted / map / filter
nums = [3, 1, 4, 1, 5, 9]
sorted(nums, key=lambda x: -x)      # reverse sort

people = [{"name": "Bob", "age": 25}, {"name": "Alice", "age": 30}]
sorted(people, key=lambda p: p["age"])

# map — apply function to each element
doubled = list(map(lambda x: x * 2, [1, 2, 3]))
# [2, 4, 6]

# filter — keep elements where function is True
evens = list(filter(lambda x: x % 2 == 0, range(10)))
# [0, 2, 4, 6, 8]

Scope

x = 10  # global

def outer():
    y = 20  # local to outer

    def inner():
        nonlocal y      # modify outer's variable
        z = 30          # local to inner
        y = 99

    inner()
    print(y)   # 99

def modify_global():
    global x
    x = 999

# LEGB rule: Local → Enclosing → Global → Built-in

Return Values

# Return multiple values (actually a tuple)
def min_max(lst):
    return min(lst), max(lst)

lo, hi = min_max([3, 1, 4, 1, 5])

# Early return
def find_first(lst, target):
    for i, v in enumerate(lst):
        if v == target:
            return i
    return -1   # not found

# Return None implicitly if no return statement
def greet(name):
    print(f"Hi {name}")

Generators

Lazy sequences — values produced on demand, don’t hold all data in memory.

def countdown(n):
    while n > 0:
        yield n
        n -= 1

for x in countdown(5):
    print(x)    # 5, 4, 3, 2, 1

# Generator expression (like list comprehension but lazy)
squares = (x**2 for x in range(1000000))
next(squares)   # 0
next(squares)   # 1

# Useful for large files
def read_lines(path):
    with open(path) as f:
        for line in f:
            yield line.strip()

Decorators

Functions that wrap other functions to add behaviour.

import time
import functools

def timer(func):
    @functools.wraps(func)    # preserve func metadata
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper

@timer
def slow_func():
    time.sleep(0.1)

slow_func()  # slow_func took 0.1001s

# Decorator with arguments
def repeat(n):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for _ in range(n):
                func(*args, **kwargs)
        return wrapper
    return decorator

@repeat(3)
def say(msg):
    print(msg)

say("hello")  # prints hello 3 times

Useful Built-in Functions

# Iteration helpers
enumerate(["a","b","c"])      # (0,'a'), (1,'b'), (2,'c')
zip([1,2,3], ["a","b","c"])   # (1,'a'), (2,'b'), (3,'c')
zip(*[[1,2],[3,4],[5,6]])      # unzip: (1,3,5), (2,4,6)

# Map / filter / reduce
list(map(str, [1, 2, 3]))          # ['1', '2', '3']
list(filter(bool, [0, 1, None, 2])) # [1, 2]

from functools import reduce
reduce(lambda a, b: a + b, [1,2,3,4])  # 10

# all / any
all([True, True, True])   # True
all([True, False, True])  # False
any([False, False, True]) # True

# sorted with complex key
sorted(["fig", "banana", "apple"], key=lambda s: s[-1])
# sorted by last character
