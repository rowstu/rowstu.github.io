File I/O

Reading Files

# Read entire file as string
with open("data.txt") as f:
    content = f.read()

# Read line by line (memory efficient)
with open("data.txt") as f:
    for line in f:
        print(line.strip())

# Read all lines into a list
with open("data.txt") as f:
    lines = f.readlines()          # includes \n
    lines = f.read().splitlines()  # strips \n

# Specify encoding
with open("data.txt", encoding="utf-8") as f:
    content = f.read()

Writing Files

# Write (overwrites existing)
with open("output.txt", "w") as f:
    f.write("Hello\n")
    f.write("World\n")

# Append
with open("log.txt", "a") as f:
    f.write("New log entry\n")

# Write multiple lines
lines = ["line1", "line2", "line3"]
with open("output.txt", "w") as f:
    f.writelines(line + "\n" for line in lines)

# File modes
# 'r'  — read (default)
# 'w'  — write (truncates existing)
# 'a'  — append
# 'x'  — exclusive create (fails if exists)
# 'b'  — binary mode (e.g. 'rb', 'wb')
# '+'  — read+write (e.g. 'r+', 'w+')

pathlib (preferred over os.path)

from pathlib import Path

p = Path("/home/user/docs/report.txt")

# Parts
p.name          # 'report.txt'
p.stem          # 'report'
p.suffix        # '.txt'
p.parent        # Path('/home/user/docs')
p.parts         # ('/', 'home', 'user', 'docs', 'report.txt')

# Navigation
home = Path.home()              # /home/user
cwd = Path.cwd()                # current directory
config = home / ".config" / "myapp"  # join with /

# Check existence
p.exists()
p.is_file()
p.is_dir()

# Read / write (no need to open())
text = p.read_text(encoding="utf-8")
p.write_text("Hello world\n")

data = p.read_bytes()
p.write_bytes(b"\x00\x01")

# Create / delete
p.mkdir(parents=True, exist_ok=True)
p.unlink()           # delete file
p.rmdir()            # delete empty dir

# List directory
for f in Path(".").iterdir():
    print(f)

# Glob patterns
for py_file in Path(".").glob("**/*.py"):
    print(py_file)

# Rename / copy
p.rename("new_name.txt")
import shutil
shutil.copy(p, "backup.txt")
shutil.copytree("src_dir", "dst_dir")

# Get file info
stat = p.stat()
stat.st_size    # file size in bytes
stat.st_mtime   # modification time

CSV

import csv

# Read
with open("data.csv", newline="") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["name"], row["age"])

# Write
data = [{"name": "Alice", "age": 30}, {"name": "Bob", "age": 25}]
with open("out.csv", "w", newline="") as f:
    writer = csv.DictWriter(f, fieldnames=["name", "age"])
    writer.writeheader()
    writer.writerows(data)

JSON

import json

# Read JSON file
with open("data.json") as f:
    data = json.load(f)

# Write JSON file
with open("out.json", "w") as f:
    json.dump(data, f, indent=2)

# String conversion
text = json.dumps({"key": "value"}, indent=2)
data = json.loads('{"key": "value"}')

# Handle non-serialisable types
from datetime import datetime
json.dumps({"date": datetime.now()}, default=str)

Error Handling for Files

from pathlib import Path

# Check before opening
p = Path("config.txt")
if p.exists():
    text = p.read_text()

# Or catch the exception
try:
    with open("config.txt") as f:
        content = f.read()
except FileNotFoundError:
    content = ""
except PermissionError:
    print("Access denied")
