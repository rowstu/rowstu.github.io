Python Basics

Variables & Types

# Dynamic typing — no declaration needed
x = 42           # int
y = 3.14         # float
name = "Alice"   # str
flag = True      # bool
nothing = None   # NoneType

# Check type
type(x)          # <class 'int'>
isinstance(x, int)  # True

# Multiple assignment
a, b, c = 1, 2, 3
a = b = c = 0

Strings

s = "hello world"

# Common methods
s.upper()           # 'HELLO WORLD'
s.capitalize()      # 'Hello world'
s.title()           # 'Hello World'
s.strip()           # remove whitespace
s.replace("o", "0") # 'hell0 w0rld'
s.split(" ")        # ['hello', 'world']
s.startswith("hel") # True
s.count("l")        # 3
len(s)              # 11

# Slicing
s[0:5]    # 'hello'
s[-5:]    # 'world'
s[::2]    # 'hlowrd'  (every 2nd char)
s[::-1]   # 'dlrow olleh' (reversed)

# f-strings (Python 3.6+)
name = "Alice"
age = 30
f"Name: {name}, Age: {age}"  # 'Name: Alice, Age: 30'
f"{3.14159:.2f}"             # '3.14'
f"{1000000:,}"               # '1,000,000'
f"{42:>10}"                  # '        42' (right-align, width 10)
f"{42:0>5}"                  # '00042' (zero-pad)

# Multiline strings
text = """
line one
line two
"""

Numbers & Operators

# Arithmetic
7 + 2    # 9
7 - 2    # 5
7 * 2    # 14
7 / 2    # 3.5  (always float)
7 // 2   # 3    (floor division)
7 % 2    # 1    (modulo)
7 ** 2   # 49   (power)

# Augmented assignment
x += 1
x -= 1
x *= 2
x //= 2

# Comparison
x == y    # equal
x != y    # not equal
x > y     # greater
x < y     # less
x >= y    # greater or equal
x <= y    # less or equal

# Logical
True and False  # False
True or False   # True
not True        # False

# Identity / membership
x is None          # True if x is None
x is not None      # True if x is not None
3 in [1, 2, 3]     # True
"a" not in "dog"   # True

# Useful builtins
abs(-5)         # 5
round(3.567, 2) # 3.57
min(3, 1, 4)    # 1
max(3, 1, 4)    # 4
sum([1,2,3])    # 6
divmod(7, 2)    # (3, 1)  — quotient and remainder
pow(2, 10)      # 1024

Control Flow

# if / elif / else
x = 42
if x > 100:
    print("big")
elif x > 10:
    print("medium")
else:
    print("small")

# Ternary (one-liner)
result = "even" if x % 2 == 0 else "odd"

# for loops
for i in range(5):        # 0, 1, 2, 3, 4
    print(i)

for i in range(2, 10, 2): # 2, 4, 6, 8
    print(i)

for item in ["a", "b", "c"]:
    print(item)

# enumerate — index + value
for i, val in enumerate(["a", "b", "c"]):
    print(i, val)   # 0 a, 1 b, 2 c

# while
x = 0
while x < 5:
    x += 1

# break / continue
for i in range(10):
    if i == 3:
        continue   # skip 3
    if i == 7:
        break      # stop at 7

Type Conversion

int("42")       # 42
int(3.9)        # 3  (truncates)
float("3.14")   # 3.14
str(42)         # '42'
bool(0)         # False
bool("")        # False
bool([])        # False
bool(None)      # False
bool(1)         # True
bool("x")       # True

list((1, 2, 3))    # [1, 2, 3]
tuple([1, 2, 3])   # (1, 2, 3)
set([1, 1, 2, 3])  # {1, 2, 3}
